stateDiagram-v2
    [*] --> Idle
    Idle --> SessionStarted: startDiagnosticSession()
    SessionStarted --> Authenticated: authenticate()
    Authenticated --> Active: openSession()
    Active --> Paused: pauseSession()
    Paused --> Active: resumeSession()
    Active --> Completed: finishSession()
    Active --> Failed: errorDetected()
    Failed --> [*]
    Completed --> [*]

Reference C++ Example

#include <iostream>
#include <string>

struct SecurityCheck {
    std::string certificate;
    std::string otaStatus;
};

int main() {
    SecurityCheck check{"Device certificate present", "OTA ready"};
    std::cout << "Certificate: " << check.certificate << "\n";
    std::cout << "OTA: " << check.otaStatus << "\n";
    return 0;
}
